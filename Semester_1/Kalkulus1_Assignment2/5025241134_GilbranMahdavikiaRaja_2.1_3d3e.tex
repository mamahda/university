        \item[(e)] $f(\frac{2}{3})$ 
        \begin{description}
            \item[$\Leftrightarrow$] $f(\frac{2}{3}) = -3\frac{2}{3}, \frac{2}{3} < 1$
            \item[$\Leftrightarrow$] $f(\frac{2}{3}) = \frac{-6}{3}, \frac{2}{3} < 1$
            \item[$\Leftrightarrow$] $f(\frac{2}{3}) = -2, \frac{2}{3} < 1$
